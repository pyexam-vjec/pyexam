with open("sample.txt", "w") as f:
    f.write("Hello World! This is Python. File handling is fun, isn't it? ")

def analyze_text(filename):
    with open(filename, "r") as file:
        text = file.read()

    words = text.split()
    word_count = len(words)
    print("Words list:", words)

    sentence_count = 0
    for ch in text:
        if ch in ".?!":
            sentence_count += 1

    upper_count = 0
    lower_count = 0
    special_count = 0

    for ch in text:
        if ch.isupper():
            upper_count += 1
        elif ch.islower():
            lower_count += 1
        elif not ch.isdigit() and not ch.isspace():
            special_count += 1

    print("Words:", word_count)
    print("Sentences:", sentence_count)
    print("Uppercase letters:", upper_count)
    print("Lowercase letters:", lower_count)
    print("Special symbols:", special_count)

filename = "sample.txt"
analyze_text(filename)
