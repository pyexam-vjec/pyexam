import matplotlib.pyplot as plt
heights = [135, 140, 145, 150, 155]
students = [4, 12, 16, 8]
midpoints = [(heights[i] + heights[i+1]) / 2 for i in range(len(heights)-1)]
plt.bar(midpoints, students, width=5, edgecolor="black", color="skyblue")
plt.xlabel("Height of Students (cm)")
plt.ylabel("Number of Students")
plt.title("Histogram of Students' Heights")
plt.grid(axis="y", linestyle="--", alpha=0.7)
plt.show()
