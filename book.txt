books = {
    "Python Programming": 5,
    "Data Science 101": 3,
    "Machine Learning Basics": 4
}

def display_books():
    print("\nCurrent Book List:")
    for title, stock in books.items():
        print(f"{title} - {stock} copies")
    print()

while True:
    print("1. Display books")
    print("2. Update stock")
    print("3. Add book")
    print("4. Delete book")
    print("5. Exit")
    choice = input("Enter your choice (1-5): ")
    
    if choice == "1":
        display_books()
    elif choice == "2":
        title = input("Enter the book title to update: ")
        if title in books:
            new_stock = int(input("Enter new stock quantity: "))
            books[title] = new_stock
            print("Stock updated successfully!")
        else:
            print("Book not found.")
    elif choice == "3":
        title = input("Enter new book title: ")
        if title in books:
            print("Book already exists!")
        else:
            stock = int(input("Enter stock quantity: "))
            books[title] = stock
            print("Book added successfully!")
    elif choice == "4":
        title = input("Enter the book title to delete: ")
        if title in books:
            del books[title]
            print("Book deleted successfully!")
        else:
            print("Book not found.")
    elif choice == "5":
        print("Exiting program.")
        break
    else:
        print("Invalid choice, try again.")
